    const getRandomElements = (arr, count, excludeCodes = []) => {

      const filtered = arr.filter(c => !excludeCodes.includes(c.code));

      const shuffled = [...filtered].sort(() => 0.5 - Math.random());

      return shuffled.slice(0, count);

    };

 

    // --- Local storage hook ---

    const useLocalStorage = (key, initialValue) => {

      const [storedValue, setStoredValue] = useState(() => {

        try {

          const item = window.localStorage.getItem(key);

          return item ? JSON.parse(item) : initialValue;

        } catch (error) {

          console.error(error);

          return initialValue;

        }

      });

 

      const setValue = (value) => {

        try {

          const valueToStore = value instanceof Function ? value(storedValue) : value;

          setStoredValue(valueToStore);

          window.localStorage.setItem(key, JSON.stringify(valueToStore));

        } catch (error) {

          console.error(error);

        }

      };

 

      return [storedValue, setValue];

    };

      const [highScores, setHighScores] = useLocalStorage('flagGameHighScores', []);

      const [showHighScores, setShowHighScores] = useState(false);

            progressBarRef.current.style.backgroundColor = '#4a6fa5';

          }

        }

      }, [timeLeft]);

 

      useEffect(() => {

          generateQuestion([]);

        }

        // Pick questionCountry from the current difficulty group

        const newQuestion = available[Math.floor(Math.random() * available.length)];

        // Pick distractors from all countries, excluding the questionCountry

        const wrongOptions = getRandomElements(countries, optionCount - 1, [newQuestion.code]);

        const allOptions = [...wrongOptions, newQuestion].sort(() => 0.5 - Math.random());

 

        setQuestionCountry(newQuestion);

        setFlagOptions(allOptions);

        setFeedback('');

        setClickedFlag(null);

        setDisableClicks(false);

      const handleFlagClick = (selectedCountry) => {

        const isCorrect = selectedCountry.code === questionCountry.code;

        setDisableClicks(true);

        setClickedFlag(selectedCountry.code);

        setTotalAnswers(prev => prev + 1);

          setScore(prev => prev + pointsEarned);

          setCorrectAnswers(prev => prev + 1);

          setCurrentStreak(prev => prev + 1);

          setStreak(prev => Math.max(prev, currentStreak + 1));

          setFeedback(`✅ Correct! ${streakBonus > 0 ? `+${streakBonus} streak bonus!` : ''}`);

          setTimeout(() => generateQuestion(), 1000);

        } else {

          setCurrentStreak(0);

          setFeedback(`❌ Incorrect. That's the flag of ${selectedCountry.name}.`);

          setTimeout(() => generateQuestion(), 1500);

        }

      };

 

      const toggleHighScores = () => {

        setShowHighScores(!showHighScores);

      };

 

      const getTimerClass = () => {

        if (timeLeft <= 10) return 'danger';

        if (timeLeft <= 20) return 'warning';

        return '';

      };

 

      const getAccuracyPercentage = () => {

        if (totalAnswers === 0) return 0;

        return Math.round((correctAnswers / totalAnswers) * 100);

      };

 

      // Welcome/Start screen

              <p>Test your knowledge of world flags!</p>

              <h2>Select Difficulty</h2>

              <div className="difficulty-selector">

                  className={`difficulty-btn ${difficulty === 'easy' ? 'active' : ''}`}

                  onClick={() => setDifficulty('easy')}

                >

                  Easy

                </button>

                <button 

                  className={`difficulty-btn ${difficulty === 'medium' ? 'active' : ''}`}

                  onClick={() => setDifficulty('medium')}

                >

                  Medium

                </button>

                <button 

                  className={`difficulty-btn ${difficulty === 'hard' ? 'active' : ''}`}

                  onClick={() => setDifficulty('hard')}

                >

                  Hard

                </button>

              </div>

              <p>

                {difficulty === 'easy' && 'Only the most recognizable flags from around the world.'}

                {difficulty === 'medium' && 'A good mix of flags from different regions.'}

                {difficulty === 'hard' && 'All flags including territories and dependencies.'}

              </p>

              <div className="button-container">

                <button className="button success" onClick={startNewGame}>

                  Start Game

                </button>

                <button className="button" onClick={toggleHighScores}>

                  High Scores

                </button>

              </div>

            </div>

          </div>

        );

      }

 

      // Game over screen

      if (gameOver) {

        return (

          <div className="container">

            <header>

              <h1>Game Over!</h1>

            </header>

            <div className="game-over">

              <div className="final-score">{score}</div>

              <p>Your final score</p>

              <div className="game-stats">

                <div className="stat">

                  <span className="stat-value">{correctAnswers}</span>

                  <span className="stat-label">Correct</span>

                </div>

                <div className="stat">

                  <span className="stat-value">{totalAnswers - correctAnswers}</span>

                  <span className="stat-label">Wrong</span>

                </div>

                <div className="stat">

                  <span className="stat-value">{getAccuracyPercentage()}%</span>

                  <span className="stat-label">Accuracy</span>

                </div>

                <div className="stat">

                  <span className="stat-value">{streak}</span>

                  <span className="stat-label">Best Streak</span>

                </div>

              </div>

              <div className="button-container">

                <button className="button success" onClick={startNewGame}>

                  Play Again

                </button>

                <button className="button" onClick={toggleHighScores}>

                  High Scores

                </button>

              </div>

            </div>

          </div>

        );

      }

 

      // Main game screen

      return (

        <div className="container">

          <header>

            <h1>Flag Guess Game</h1>

          </header>

          <div className="game-info">

            <div className="game-stats">

              <div className="stat">

                <span className="stat-value">{score}</span>

                <span className="stat-label">Score</span>

              </div>

              <div className="stat">

                <span className="stat-value" id="timer" className={getTimerClass()}>{timeLeft}</span>

                <span className="stat-label">Seconds</span>

              </div>

              <div className="stat">

                <span className="stat-value">{currentStreak}</span>

                <span className="stat-label">Streak</span>

              </div>

            </div>

            <label>

              <input 

                type="checkbox" 

                checked={showCountryNames} 

                onChange={() => setShowCountryNames(!showCountryNames)}

              /> Show country names

            </label>

          </div>

          <div className="progress-container">

            <div 

              className="progress-bar" 

              ref={progressBarRef}

              style={{ width: `${(timeLeft / difficultySettings[difficulty].time) * 100}%` }}

            ></div>

          </div>

          {questionCountry && (

            <>

              <div className="question">

                Which is the flag of {questionCountry.name}?

              </div>

              <div className="flag-options">

                {flagOptions.map((country) => (

                  <div 

                    key={country.code} 

                    className="flag-option"

                    onClick={() => handleFlagClick(country)}

                  >

                    <img

                      className={`flag-img ${clickedFlag === country.code && country.code === questionCountry.code ? 'correct' : 

                                           clickedFlag === country.code && country.code !== questionCountry.code ? 'incorrect' : ''}`}

                      src={`https://flagcdn.com/w320/${country.code}.png`}

                      alt={country.name}

                    />

                    {showCountryNames && (

                      <div className="flag-name">{country.name}</div>

                    )}

                  </div>

                ))}

              </div>

              <div className="feedback">{feedback}</div>

            </>

          )}

          {/* High Scores Modal */}

          <div className={`modal ${showHighScores ? 'active' : ''}`}>

            <div className="modal-content">

              <div className="modal-header">

                <h2>High Scores</h2>

                <button className="modal-close" onClick={toggleHighScores}>×</button>

              </div>

              {highScores.length > 0 ? (

                <ul className="high-scores">

                  {highScores.map((entry, index) => (

                    <li key={index} className="high-score-item">

                      <span className="high-score-position">{index + 1}</span>

                      <div className="high-score-info">

                        <div>{new Date(entry.date).toLocaleDateString()}</div>

                        <div className="high-score-difficulty">

                          {entry.difficulty.charAt(0).toUpperCase() + entry.difficulty.slice(1)} · {entry.accuracy}% accuracy

                        </div>

                      </div>

                      <span className="high-score-score">{entry.score}</span>

                    </li>

                  ))}

                </ul>

              ) : (

                <p>No high scores yet. Start playing to set some records!</p>

              )}

              <div className="button-container">

                <button className="button" onClick={toggleHighScores}>Close</button>

              </div>

            </div>

          </div>

        </div>

      );

    };

 

    ReactDOM.render(<App />, document.getElementById('root'));

  </script>
